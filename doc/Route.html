def initialize(context, from, to)
  @nodes = [from, to]
  @nodes.each{|n| n.add_route self}
  @length = (@nodes[0].pos - @nodes[1].pos)[0] + (@nodes[0].pos - @nodes[1].pos)[1]
  @pheromone = 30
end
