def show_menu()
  flow  width:140, height600600 do
    background black
    flow height:25 do
      para fg('Iterations: ',white),     width9090, height2020
      @menu_interations = edit_line                width:50, height2020
      @menu_interations.text = '1000'
      
      button 'Start!',                             width1.01.0, height2020 do
        start(@menu_interations.text.to_i)
      end
      para
      @menu_turn = para fg(strong('Turn: '),white),             width9090, height2020
      @menu_population = para fg(strong('Pop.: '),white),       width9090, height2020
      @menu_food_collected = para fg(strong('Food:  '),white),  width9090, height2020
    end
  end
end
